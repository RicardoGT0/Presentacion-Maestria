%	Headline
	
\setbeamertemplate{headline}
{
	\linethickness{0pt}
	\begin{beamercolorbox}{section in head/foot}
		\vskip 0.05cm
		\insertnavigation{0.3\paperwidth}
		\vskip 0.05cm
	\end{beamercolorbox}
}



%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%	Footline

\setbeamertemplate{footline}
{	
 	\linethickness{0pt}

  	\begin{beamercolorbox}[leftskip=0.3cm,wd=\paperwidth,ht=0.3cm,sep=0.03cm]{section in head/foot}
		\insertshortauthor~|~\insertshorttitle
	    \hfill
    	\insertframenumber/\inserttotalframenumber
   	\end{beamercolorbox}
}



%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%	Frametitle

\setbeamertemplate{frametitle}
{
\nointerlineskip

    \hspace{1ex}\insertframetitle
\end{beamercolorbox}%
}
